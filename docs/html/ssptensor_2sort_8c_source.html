39 static void spt_QuickSortIndex(sptSemiSparseTensor *tsr, size_t l, size_t r, sptScalar buffer[]) {
40  size_t i, j, p;
41  if(r-l < 2) {
42  return;
43  }
44  p = (l+r) / 2;
45  for(i = l, j = r-1; ; ++i, --j) {
46  while(spt_SemiSparseTensorCompareIndices(tsr, i, tsr, p) < 0) {
47  ++i;
48  }
49  while(spt_SemiSparseTensorCompareIndices(tsr, p, tsr, j) < 0) {
50  --j;
51  }
52  if(i >= j) {
53  break;
54  }
55  spt_SwapValues(tsr, i, j, buffer);
56  if(i == p) {
57  p = j;
58  } else if(j == p) {
59  p = i;
60  }
61  }
62  spt_QuickSortIndex(tsr, l, i, buffer);
63  spt_QuickSortIndex(tsr, i, r, buffer);
64 }
