32 int sptNewSemiSparseTensor(sptSemiSparseTensor *tsr, size_t nmodes, size_t mode, const size_t ndims[]) {
33  size_t i;
34  int result;
35  if(nmodes < 2) {
36  spt_CheckError(SPTERR_SHAPE_MISMATCH, "SspTns New", "nmodes < 2");
37  }
38  tsr->nmodes = nmodes;
39  tsr->ndims = malloc(nmodes * sizeof *tsr->ndims);
40  spt_CheckOSError(!tsr->ndims, "SspTns New");
41  memcpy(tsr->ndims, ndims, nmodes * sizeof *tsr->ndims);
42  tsr->mode = mode;
43  tsr->nnz = 0;
44  tsr->inds = malloc(nmodes * sizeof *tsr->inds);
45  spt_CheckOSError(!tsr->inds, "SspTns New");
46  for(i = 0; i < nmodes; ++i) {
47  result = sptNewSizeVector(&tsr->inds[i], 0, 0);
48  spt_CheckError(result, "SspTns New", NULL);
49  }
50  tsr->stride = ((ndims[mode]-1)/8+1)*8;
51  result = sptNewMatrix(&tsr->values, 0, tsr->stride);
52  spt_CheckError(result, "SspTns New", NULL);
53  return 0;
54 }
