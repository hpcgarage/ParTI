38 static void spt_QuickSortIndex(sptSparseTensor *tsr, size_t l, size_t r) {
39  size_t i, j, p;
40  if(r-l < 2) {
41  return;
42  }
43  p = (l+r) / 2;
44  for(i = l, j = r-1; ; ++i, --j) {
45  while(spt_SparseTensorCompareIndices(tsr, i, tsr, p) < 0) {
46  ++i;
47  }
48  while(spt_SparseTensorCompareIndices(tsr, p, tsr, j) < 0) {
49  --j;
50  }
51  if(i >= j) {
52  break;
53  }
54  spt_SwapValues(tsr, i, j);
55  if(i == p) {
56  p = j;
57  } else if(j == p) {
58  p = i;
59  }
60  }
61  spt_QuickSortIndex(tsr, l, i);
62  spt_QuickSortIndex(tsr, i, r);
63 }
