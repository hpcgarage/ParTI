30 int spt_SemiSparseTensorMergeValues(sptSemiSparseTensor *tsr) {
31  int result;
32  size_t i;
33  sptSizeVector collided;
34  sptScalar *buffer;
35 
36  if(tsr->nnz == 0) {
37  return 0;
38  }
39 
40  buffer = malloc(tsr->stride * sizeof (sptScalar));
41  spt_CheckOSError(!buffer, "SspTns Merge");
42 
43  result = sptNewSizeVector(&collided, 0, 0);
44  if(result) {
45  free(buffer);
46  spt_CheckError(result, "SspTns Merge", NULL);
47  }
48 
49  for(i = 0; i < tsr->nnz-1; ++i) {
50  // If two nnz has the same indices
51  if(spt_SemiSparseTensorCompareIndices(tsr, i, tsr, i+1) == 0) {
52  size_t col;
53  for(col = 0; col < tsr->stride; ++col) {
54  // Add them together
55  tsr->values.values[(i+1)*tsr->stride + col] += tsr->values.values[i*tsr->stride + col];
56  }
57  sptAppendSizeVector(&collided, i);
58  }
59  }
60 
61  // Swap the last invalidated item with the last item
62  i = collided.len;
63  while(i) {
64  --i;
65  assert(tsr->nnz != 0);
66  spt_SwapValues(tsr, collided.data[i], tsr->nnz-1, buffer);
67  --tsr->nnz;
68  }
69 
70  // Make sure all Vectors and Matrices have correct sizes
71  for(i = 0; i < tsr->nmodes; ++i) {
72  if(i != tsr->mode) {
73  tsr->inds[i].len = tsr->nnz;
74  }
75  }
76  tsr->values.nrows = tsr->nnz;
77 
78  sptFreeSizeVector(&collided);
79  free(buffer);
80 
81  result = sptSemiSparseTensorSortIndex(tsr);
82  spt_CheckError(result, "SspTns Merge", NULL);
83  return 0;
84 }
